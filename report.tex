\begin{align*}
  \begin{split}
  M(x', t'; x, t) &= \prescript{}{H}{\braket{x',t'|x, t}_{H}}
  \end{split} \\
  \begin{split}
  &= \braket{x'|\exp{\left[\frac{-i}{\hbar}\hat{H}(t-t')\right]}|x}
